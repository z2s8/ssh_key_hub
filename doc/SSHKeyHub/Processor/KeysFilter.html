def reject_weak
  @credentials.each do |user, keys|
    keys.delete_if do |key|
      type, bits = SSHKeyHub::Processor::KeyProcessor.new.key_type_and_bits(key)
      puts "testing #{type} with #{bits}"
      case type
      when :DSA
        true
      when :RSA
        bits < 4096
      when :EC
        bits < 256
      else
        puts "[KeysFilter] Allowing UNRECOGNIZED key #{key} for #{user}"
        false
      end
    end
  end
  @credentials
end
